#include<stdio.h>
int fact(int n){
  int i,factorial=1;
   for (i=1;i<=n;i++){
    factorial*=i;
}
return factorial;
}
int main(){
int num;
printf("enter a non negative number");
scanf("%d",&num);
if (num<0){
printf("factorial is not defined for negative number\n");
   }else{
int result=fact(num);
printf("factorial of%d is%d\n",num.result);
    }\
   return 0;
}s
